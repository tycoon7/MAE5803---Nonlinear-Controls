mu = [-0.2 0 0.2];
for i = 1:length(mu)
    figure()
    hold on
    for x1 = -1.5:.5:1.5
        for x2 = -1.5:.5:1.5
            tspan = [0 3];
            x0 = [x1; x2];
            [t,x] = ode45(@P3stateEqn,tspan,x0,[],mu(i));
            h = plot(x(:,1),x(:,2));
            c = get(h,'color');

    xlabel('$x_1$')
    ylabel('$x_2$')
